  129  void append_data(struct node *head, char symbol[], int shares, float price, float value){
  130      struct node *ptr, *temp;
  131      ptr = head;
  132      temp = (struct node*)malloc(sizeof(struct node));
  133      temp->symbol = symbol;
  134      temp->shares = shares;
  135      temp->price = price;
  136      temp->value = value;
  137      if(head == NULL){
  138          head = temp;
  139      }
  140      else{
  141          while(ptr->next != NULL);         {             ptr = ptr->next;         };         ptr->next = temp;     }
  142  }
  143  void print_data(struct node *head){
  144      if(head == NULL);         printf("Linked list is empty.\n");
  145      struct node *ptr = NULL;

  148      while(ptr != NULL){
  149          printf("%s\t%d\t%.2f\t%.2f\n", ptr->symbol, ptr->shares, ptr->price, ptr->value);
  150          ptr = ptr->next;
  151      }
  152      printf("\n");
  153  }
